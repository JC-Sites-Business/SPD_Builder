Sub Generate_Quote()

End Sub

Sub Payment_Schedule()
    Dim firstRow, currentRow, lastRow, reqLastRow As Long
    Dim currentRange As Range
    Dim i As Integer
    Dim tempWorkItem As String
    Dim tempLabour As Integer
    Dim tempDays As Double
    
    Sheets("Quote").Select
    firstRow = Cells(Rows.Count, "B").End(xlUp).Row + 2
    
    Range("B" & firstRow & ":E" & firstRow).Select
    Selection.Merge
    Selection.Font.Size = 18
    Selection.Font.Bold = True
    Selection.HorizontalAlignment = xlCenter
    ActiveCell.FormulaR1C1 = "Payment Shedule"
    
    firstRow = firstRow + 1
    Range("B15:E15").Select
    Selection.Copy
    Range("B" & firstRow).Select
    ActiveSheet.Paste
    Range("D" & firstRow).Select
    ActiveCell.FormulaR1C1 = "Est Day"
    Range("E" & firstRow).Select
    ActiveCell.FormulaR1C1 = "(£)"
    
    firstRow = firstRow + 1
    Range("B" & firstRow).Select
    ActiveCell.FormulaR1C1 = "Materials"
    Range("D" & firstRow).Select
    ActiveCell.FormulaR1C1 = "0"
    Range("E" & firstRow).Select
    ActiveCell.Formula = "=Requirements!$G$4"
    Selection.Copy
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks:=False, Transpose:=False
    
    Sheets("Requirements").Select
    tempDays = 0
    reqLastRow = Cells(Rows.Count, "D").End(xlUp).Row
    currentRow = firstRow
    For i = 7 To reqLastRow
        If Cells(i, 4).Value = "Yes" Then
            If tempWorkItem = "" Then
                tempWorkItem = Cells(i, 3).Value
                tempDays = IIf(tempDays = 0, Cells(i, 5).Value, tempDays + Cells(i, 5).Value)
                tempLabour = Cells(i, 6).Value
            Else
                tempWorkItem = tempWorkItem & ", " & Cells(i, 3).Value
                tempDays = tempDays + Cells(i, 5).Value
                tempLabour = tempLabour + Cells(i, 6).Value
            End If
            
            Sheets("Quote").Select
            currentRow = currentRow + 1
            Range("B" & currentRow).Select
            ActiveCell.FormulaR1C1 = tempWorkItem
            Range("D" & currentRow).Select
            ActiveCell.FormulaR1C1 = Application.WorksheetFunction.RoundUp(tempDays, 0)
            Range("E" & currentRow).Select
            ActiveCell.Formula = tempLabour
            tempWorkItem = ""
            tempLabour = 0
            Sheets("Requirements").Select
            
        Else
            If tempWorkItem = "" Then
                tempWorkItem = Cells(i, 3).Value
                tempDays = IIf(tempDays = 0, Cells(i, 5).Value, tempDays + Cells(i, 5).Value)
                tempLabour = Cells(i, 6).Value
            Else
                tempWorkItem = tempWorkItem & ", " & Cells(i, 3).Value
                tempDays = tempDays + Cells(i, 5).Value
                tempLabour = tempLabour + Cells(i, 6).Value
            End If
        End If
    Next i
    
    Sheets("Quote").Select
    lastRow = Cells(Rows.Count, "B").End(xlUp).Row
    Range("B" & firstRow & ":C" & lastRow).Select
    Selection.BorderAround xlContinuous, xlMedium, , -10477568
    With Selection.Borders(xlInsideHorizontal)
        .LineStyle = xlContinuous
        .Color = -10477568
        .Weight = xlThin
    End With
    Range("D" & firstRow & ":E" & lastRow).Select
    Selection.BorderAround xlContinuous, xlMedium, , -10477568
    With Selection.Borders(xlInsideVertical)
        .LineStyle = xlContinuous
        .Color = -10477568
        .TintAndShade = 0
        .Weight = xlMedium
    End With
    With Selection.Borders(xlInsideHorizontal)
        .LineStyle = xlContinuous
        .Color = -10477568
        .TintAndShade = 0
        .Weight = xlThin
    End With
    
    lastRow = lastRow + 2
    Range("B" & lastRow).Select
    ActiveCell.FormulaR1C1 = "Total"
    Range("E" & lastRow).Select
    ActiveCell.FormulaR1C1 = WorksheetFunction.Sum(Range("E" & firstRow & ":E" & lastRow))
    Range("B" & lastRow & ":E" & lastRow).Select
    With Selection.Font
        .FontStyle = "Bold"
    End With
    With Selection.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .Color = -10477568
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeBottom)
        .LineStyle = xlDouble
        .Color = -10477568
        .Weight = xlThick
    End With
    
    Range("E" & firstRow & ":E" & lastRow).Select
    Selection.NumberFormat = "$#,##0"
End Sub

Sub Export()
    Dim SheetNames As Variant
    Dim SavePath As String
    Dim quoteLastRow, breakdownLastRow As Long
    
    Sheets("Quote").Activate
    quoteLastRow = Cells(Rows.Count, "B").End(xlUp).Row
    Sheets("Quote_Breakdown").Activate
    breakdownLastRow = Cells(Rows.Count, "B").End(xlUp).Row
    
    SheetNames = Array("Quote", "Quote_Breakdown")
    SavePath = ThisWorkbook.Path & "\" & Sheets("Quote").Range("C5").Value & ".pdf"
    
    Sheets("Quote").PageSetup.PrintArea = ""
    With Sheets("Quote").PageSetup
        .PrintArea = "$B$1:$E$" & quoteLastRow
        .Zoom = False
        .FitToPagesWide = 1
        .FitToPagesTall = 1
    End With
    
    Sheets("Quote_Breakdown").PageSetup.PrintArea = ""
    With Sheets("Quote_Breakdown").PageSetup
        .PrintArea = "$B$1:$D$" & breakdownLastRow
        .Zoom = False
        .FitToPagesWide = 1
        .FitToPagesTall = 1
    End With
    
    Sheets(SheetNames).Select
    ActiveSheet.ExportAsFixedFormat _
        Type:=xlTypePDF, _
        fileName:=SavePath, _
        Quality:=xlQualityStandard, _
        IncludeDocProperties:=True, _
        IgnorePrintAreas:=False, _
        OpenAfterPublish:=True
    
    Sheets(SheetNames(0)).Select
End Sub
Sub Save()
    ThisWorkbook.SaveCopyAs ThisWorkbook.Path & "\" & Sheets("Customer").Range("C4").Value & ".xlsm"
    
    MsgBox "PDF and Excel Backup Created Successfully!", vbInformation
End Sub
